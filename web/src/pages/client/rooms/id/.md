"use client";

import { Card, CardHeader, CardTitle, CardContent } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import {
  CheckCircle,
  XCircle,
  Bath,
  BedDouble,
  CookingPot,
  Sofa,
  ShieldCheck,
  Home,
} from "lucide-react";

type RoomFeature = {
  id: string;
  category?: string | null;
  name: string;
  value?: string | null;
};

interface RoomFeaturesProps {
  features: RoomFeature[];
}

const categoryIcons: Record<string, React.ElementType> = {
  bathroom: Bath,
  bedroom: BedDouble,
  kitchen: CookingPot,
  livingroom: Sofa,
  hygiene: ShieldCheck,
  accessibility: Home,
};

export default function RoomFeatures({ features }: RoomFeaturesProps) {
  if (!features?.length) return null;

  // Group features by category
  const grouped = features.reduce<Record<string, RoomFeature[]>>(
    (acc, feature) => {
      const key = feature.category?.toLowerCase() || "other";
      if (!acc[key]) acc[key] = [];
      acc[key].push(feature);
      return acc;
    },
    {}
  );

  return (
    <div className="flex flex-col gap-2">
      <h2>Room features</h2>
      <div className="grid gap-4 sm:gap-6 sm:grid-cols-2 lg:grid-cols-3">
        {Object.entries(grouped).map(([category, items]) => {
          const Icon = categoryIcons[category] || Home;

          // If category has only one feature, render compact version
          if (items.length === 1) {
            const single = items[0];
            const displayValue = single.value || single.name;

            return (
              <Card
                key={category}
                className="rounded-2xl shadow-sm border border-border/60 hover:shadow-md transition-all duration-200 w-fit flex flex-row gap-2 p-2 items-center"
              >
                <Icon className="w-5 h-5 text-muted-foreground shrink-0" />
                <CardTitle className="capitalize text-base font-semibold truncate">
                  - {displayValue}
                </CardTitle>
              </Card>
            );
          }

          // Otherwise render normal list
          return (
            <Card
              key={category}
              className="rounded-2xl shadow-sm border border-border/60 hover:shadow-md transition-all duration-200"
            >
              <CardHeader className="flex flex-row items-center gap-2 pb-2">
                <Icon className="w-5 h-5 text-muted-foreground shrink-0" />
                <CardTitle className="capitalize text-base sm:text-lg font-semibold truncate">
                  {category.replace(/_/g, " ")}
                </CardTitle>
              </CardHeader>

              <CardContent className="divide-y divide-border/40">
                {items.map((feature) => {
                  const value = feature.value?.toLowerCase();
                  const isTrue = value === "true";
                  const isFalse = value === "false";

                  return (
                    <div
                      key={feature.id}
                      className="flex flex-wrap items-center justify-between py-2 text-sm"
                    >
                      <span className="capitalize text-muted-foreground max-w-[60%] truncate">
                        {feature.name}
                      </span>
                      <div className="flex items-center gap-1 shrink-0">
                        {isTrue ? (
                          <CheckCircle className="text-green-500 w-4 h-4" />
                        ) : isFalse ? (
                          <XCircle className="text-red-500 w-4 h-4" />
                        ) : (
                          <Badge
                            variant="secondary"
                            className="text-[11px] px-2 py-0.5"
                          >
                            {feature.value || "—"}
                          </Badge>
                        )}
                      </div>
                    </div>
                  );
                })}
              </CardContent>
            </Card>
          );
        })}
      </div>
    </div>
  );
}
